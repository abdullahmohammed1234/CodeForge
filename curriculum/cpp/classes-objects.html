Classes and Objects

Classes and objects are fundamental concepts in object-oriented programming. A class is a blueprint for creating objects, while an object is an instance of a class. C++ provides powerful features for defining classes, creating objects, and implementing object-oriented principles.

What are Classes and Objects?

  • Class: A user-defined data type that encapsulates data and functions
  • Object: An instance of a class in memory
  • Encapsulation: Bundling data and methods that operate on that data
  • Abstraction: Hiding implementation details and showing only essential features
  • Modularity: Breaking down complex systems into manageable components

Defining a Class

#include <iostream>
#include <string>

class Person {
private:
    // Private data members
    std::string name;
    int age;
    
public:
    // Constructor
    Person(std::string n, int a) : name(n), age(a) {}
    
    // Destructor
    ~Person() {
        std::cout << "Person " << name << " destroyed" << std::endl;
    }
    
    // Public member functions
    void set_name(std::string n) {
        name = n;
    }
    
    std::string get_name() const {
        return name;
    }
    
    void set_age(int a) {
        if (a >= 0) {
            age = a;
        }
    }
    
    int get_age() const {
        return age;
    }
    
    void display() const {
        std::cout << "Name: " << name << ", Age: " << age << std::endl;
    }
};

int main() {
    // Create objects
    Person person1("Alice", 25);
    Person person2("Bob", 30);
    
    // Use objects
    person1.display();
    person2.display();
    
    // Modify objects
    person1.set_age(26);
    person1.display();
    
    return 0;
}

Access Specifiers

  • public: Members accessible from anywhere
  • private: Members accessible only within the class
  • protected: Members accessible within the class and derived classes
class Example {
private:
    int private_var;     // Only accessible within this class
    
protected:
    int protected_var;   // Accessible in this class and derived classes
    
public:
    int public_var;      // Accessible from anywhere
    
    void public_method() {
        private_var = 10;    // OK - same class
        protected_var = 20;  // OK - same class
        public_var = 30;     // OK - same class
    }
};

int main() {
    Example obj;
    // obj.private_var = 1;    // Error - private
    // obj.protected_var = 2;  // Error - protected
    obj.public_var = 3;       // OK - public
    
    return 0;
}

Constructors and Destructors

class Rectangle {
private:
    double width, height;
    
public:
    // Default constructor
    Rectangle() : width(0), height(0) {
        std::cout << "Default constructor called" << std::endl;
    }
    
    // Parameterized constructor
    Rectangle(double w, double h) : width(w), height(h) {
        std::cout << "Parameterized constructor called" << std::endl;
    }
    
    // Copy constructor
    Rectangle(const Rectangle& other) : width(other.width), height(other.height) {
        std::cout << "Copy constructor called" << std::endl;
    }
    
    // Destructor
    ~Rectangle() {
        std::cout << "Destructor called" << std::endl;
    }
    
    double area() const {
        return width * height;
    }
};

int main() {
    Rectangle rect1;              // Default constructor
    Rectangle rect2(5.0, 3.0);    // Parameterized constructor
    Rectangle rect3 = rect2;      // Copy constructor
    
    std::cout << "Area: " << rect2.area() << std::endl;
    
    return 0;  // Destructors called here
}

Member Functions

class BankAccount {
private:
    std::string account_number;
    double balance;
    
public:
    // Constructor
    BankAccount(std::string acc_num, double initial_balance = 0.0)
        : account_number(acc_num), balance(initial_balance) {}
    
    // Const member function - doesn't modify object
    double get_balance() const {
        return balance;
    }
    
    // Non-const member function
    void deposit(double amount) {
        if (amount > 0) {
            balance += amount;
        }
    }
    
    // Member function with reference parameter
    bool withdraw(double amount) {
        if (amount > 0 && amount <= balance) {
            balance -= amount;
            return true;
        }
        return false;
    }
    
    // Static member function
    static double get_interest_rate() {
        return 0.05;  // 5% interest
    }
    
    // Friend function declaration
    friend void display_account(const BankAccount& account);
};

// Friend function definition
void display_account(const BankAccount& account) {
    std::cout << "Account: " << account.account_number
              << ", Balance: $" << account.balance << std::endl;
}

int main() {
    BankAccount account("123456789", 1000.0);
    
    account.deposit(500.0);
    account.withdraw(200.0);
    
    display_account(account);
    
    std::cout << "Interest rate: " << BankAccount::get_interest_rate() * 100 << "%" << std::endl;
    
    return 0;
}

Static Members

class Counter {
private:
    static int count;  // Static data member
    int id;
    
public:
    Counter() {
        count++;
        id = count;
    }
    
    static int get_count() {  // Static member function
        return count;
    }
    
    int get_id() const {
        return id;
    }
};

// Initialize static member
int Counter::count = 0;

int main() {
    Counter c1, c2, c3;
    
    std::cout << "Total objects: " << Counter::get_count() << std::endl;
    std::cout << "Object IDs: " << c1.get_id() << ", "
              << c2.get_id() << ", " << c3.get_id() << std::endl;
    
    return 0;
}

The 'this' Pointer

class Point {
private:
    double x, y;
    
public:
    Point(double x = 0, double y = 0) : x(x), y(y) {}
    
    // Method chaining using 'this'
    Point& set_x(double new_x) {
        this->x = new_x;
        return *this;
    }
    
    Point& set_y(double new_y) {
        this->y = new_y;
        return *this;
    }
    
    void display() const {
        std::cout << "(" << this->x << ", " << this->y << ")" << std::endl;
    }
};

int main() {
    Point p;
    
    // Method chaining
    p.set_x(3.0).set_y(4.0).display();
    
    return 0;
}

Object Composition

class Engine {
private:
    int horsepower;
    
public:
    Engine(int hp) : horsepower(hp) {}
    int get_horsepower() const { return horsepower; }
};

class Car {
private:
    std::string model;
    Engine engine;  // Composition - Car has an Engine
    
public:
    Car(std::string m, int hp) : model(m), engine(hp) {}
    
    void display() const {
        std::cout << "Model: " << model << std::endl;
        std::cout << "Engine: " << engine.get_horsepower() << " HP" << std::endl;
    }
};

int main() {
    Car my_car("Mustang", 450);
    my_car.display();
    
    return 0;
}

Mini Project

Create a library management system using classes and objects:

  • Define a Book class with title, author, ISBN, and availability status
  • Create a Library class that manages a collection of books
  • Implement methods to add books, search by title/author, and check out/return books
  • Use proper encapsulation with private data members
  • Bonus: Add a User class and implement user borrowing limits

Here's a starting template:

#include <iostream>
#include <string>
#include <vector>

class Book {
private:
    std::string title;
    std::string author;
    std::string isbn;
    bool available;
    
public:
    Book(std::string t, std::string a, std::string i)
        : title(t), author(a), isbn(i), available(true) {}
    
    std::string get_title() const { return title; }
    std::string get_author() const { return author; }
    std::string get_isbn() const { return isbn; }
    bool is_available() const { return available; }
    
    void check_out() { available = false; }
    void return_book() { available = true; }
    
    void display() const {
        std::cout << "Title: " << title << std::endl;
        std::cout << "Author: " << author << std::endl;
        std::cout << "ISBN: " << isbn << std::endl;
        std::cout << "Status: " << (available ? "Available" : "Checked out") << std::endl;
    }
};

class Library {
private:
    std::vector<Book> books;
    
public:
    void add_book(const Book& book) {
        books.push_back(book);
    }
    
    void display_books() const {
        std::cout << "Library Collection:" << std::endl;
        for (const auto& book : books) {
            book.display();
            std::cout << "---" << std::endl;
        }
    }
    
    Book* find_book_by_title(const std::string& title) {
        for (auto& book : books) {
            if (book.get_title() == title) {
                return &book;
            }
        }
        return nullptr;
    }
    
    void check_out_book(const std::string& title) {
        Book* book = find_book_by_title(title);
        if (book && book->is_available()) {
            book->check_out();
            std::cout << "Successfully checked out: " << title << std::endl;
        } else {
            std::cout << "Book not available or not found: " << title << std::endl;
        }
    }
    
    void return_book(const std::string& title) {
        Book* book = find_book_by_title(title);
        if (book && !book->is_available()) {
            book->return_book();
            std::cout << "Successfully returned: " << title << std::endl;
        } else {
            std::cout << "Book was not checked out or not found: " << title << std::endl;
        }
    }
};

int main() {
    Library library;
    
    // Add some books
    library.add_book(Book("The C++ Programming Language", "Bjarne Stroustrup", "978-0321563842"));
    library.add_book(Book("Clean Code", "Robert C. Martin", "978-0132350884"));
    library.add_book(Book("Design Patterns", "Gang of Four", "978-0201633610"));
    
    // Display all books
    library.display_books();
    
    // Check out a book
    library.check_out_book("Clean Code");
    
    // Try to check out the same book again
    library.check_out_book("Clean Code");
    
    // Return the book
    library.return_book("Clean Code");
    
    return 0;
}
