C++ Numbers

Numbers are fundamental to programming. In C++, you work with different types of numbers including integers and floating-point numbers. Understanding how to handle numbers, perform calculations, and manage precision is crucial for writing effective C++ programs.

Number Literals

Number literals are constant values written directly in code:

Integer Literals

// Decimal (base 10)
int decimal = 42;

// Octal (base 8) - starts with 0
int octal = 052;  // 42 in decimal

// Hexadecimal (base 16) - starts with 0x or 0X
int hex = 0x2A;   // 42 in decimal
int hex2 = 0X2a;  // Also valid

// Binary (C++14) - starts with 0b or 0B
int binary = 0b101010;  // 42 in decimal

// Long integers - end with L or l
long big_num = 123456789L;

// Unsigned - end with U or u
unsigned int positive = 42U;

Floating-Point Literals

// Double precision
double pi = 3.14159;

// Float - end with F or f
float gravity = 9.81f;

// Scientific notation
double avogadro = 6.022e23;  // 6.022 × 10^23
double small = 1.23e-4;     // 1.23 × 10^-4

// Long double - end with L or l
long double precise = 3.141592653589793L;

Arithmetic Operations

C++ supports standard arithmetic operations:

Basic Operations

#include 

int main() {
    int a = 10, b = 3;
    
    std::cout << "Addition: " << a + b << std::endl;       // 13
    std::cout << "Subtraction: " << a - b << std::endl;    // 7
    std::cout << "Multiplication: " << a * b << std::endl; // 30
    std::cout << "Division: " << a / b << std::endl;       // 3 (integer division)
    std::cout << "Modulus: " << a % b << std::endl;       // 1
    
    return 0;
}

Floating-Point Division

double x = 10.0, y = 3.0;
std::cout << "Float division: " << x / y << std::endl;  // 3.333...

// Mixed types
int int_num = 5;
double float_num = 2.0;
double result = int_num / float_num;  // 2.5

Assignment Operators

int x = 5;

// Compound assignment
x += 3;  // x = x + 3;  -> 8
x -= 2;  // x = x - 2;  -> 6
x *= 4;  // x = x * 4;  -> 24
x /= 3;  // x = x / 3;  -> 8
x %= 5;  // x = x % 5;  -> 3

Increment and Decrement

int x = 5;

// Post-increment
int y = x++;  // y = 5, x = 6

// Pre-increment
int z = ++x;  // z = 7, x = 7

// Post-decrement
y = x--;      // y = 7, x = 6

// Pre-decrement
z = --x;      // z = 5, x = 5

Mathematical Functions

C++ provides mathematical functions through <cmath> header:

#include <iostream>
#include <cmath>

int main() {
    double x = 4.0, y = -3.14;
    
    std::cout << "Square root: " << sqrt(x) << std::endl;     // 2
    std::cout << "Power: " << pow(x, 2) << std::endl;         // 16
    std::cout << "Absolute value: " << abs(y) << std::endl;   // 3.14
    std::cout << "Ceiling: " << ceil(3.1) << std::endl;      // 4
    std::cout << "Floor: " << floor(3.9) << std::endl;       // 3
    std::cout << "Round: " << round(3.5) << std::endl;       // 4
    
    // Trigonometric functions (radians)
    std::cout << "Sin(π/2): " << sin(M_PI/2) << std::endl; // 1
    std::cout << "Cos(0): " << cos(0) << std::endl;         // 1
    
    return 0;
}

Number Formatting

Use <iomanip> for formatted output:

#include <iostream>
#include <iomanip>

int main() {
    double pi = 3.14159265359;
    
    // Fixed decimal places
    std::cout << std::fixed << std::setprecision(2) << pi << std::endl;  // 3.14
    std::cout << std::fixed << std::setprecision(4) << pi << std::endl;  // 3.1416
    
    // Scientific notation
    std::cout << std::scientific << pi << std::endl;  // 3.141593e+00
    
    // Field width
    std::cout << std::setw(10) << 42 << std::endl;  // Right-aligned
    
    return 0;
}

Random Numbers

C++11 introduced better random number generation:

#include <iostream>
#include <random>
#include <ctime>

int main() {
    // Seed the random number generator
    std::random_device rd;
    std::mt19937 gen(rd());
    
    // Generate random integers
    std::uniform_int_distribution<> int_dist(1, 100);
    std::cout << "Random int: " << int_dist(gen) << std::endl;
    
    // Generate random doubles
    std::uniform_real_distribution<> real_dist(0.0, 1.0);
    std::cout << "Random double: " << real_dist(gen) << std::endl;
    
    return 0;
}

Precision and Overflow

Be aware of precision limits and overflow:

#include <iostream>
#include <limits>

int main() {
    // Integer overflow
    int max_int = std::numeric_limits<int>::max();
    std::cout << "Max int: " << max_int << std::endl;
    std::cout << "Max int + 1: " << max_int + 1 << std::endl;  // Overflow!
    
    // Floating-point precision
    float f1 = 0.1f;
    float f2 = 0.2f;
    float sum = f1 + f2;
    std::cout << "0.1f + 0.2f = " << sum << std::endl;  // May not be exactly 0.3
    
    // Double has more precision
    double d1 = 0.1;
    double d2 = 0.2;
    double dsum = d1 + d2;
    std::cout << "0.1 + 0.2 = " << dsum << std::endl;
    
    return 0;
}

Example Program

#include <iostream>
#include <cmath>
#include <iomanip>

int main() {
    // Calculator program
    double num1, num2;
    char operation;
    
    std::cout << "Enter first number: ";
    std::cin >> num1;
    
    std::cout << "Enter operation (+, -, *, /, ^): ";
    std::cin >> operation;
    
    std::cout << "Enter second number: ";
    std::cin >> num2;
    
    double result;
    switch(operation) {
        case '+':
            result = num1 + num2;
            break;
        case '-':
            result = num1 - num2;
            break;
        case '*':
            result = num1 * num2;
            break;
        case '/':
            if (num2 != 0) {
                result = num1 / num2;
            } else {
                std::cout << "Error: Division by zero!" << std::endl;
                return 1;
            }
            break;
        case '^':
            result = pow(num1, num2);
            break;
        default:
            std::cout << "Invalid operation!" << std::endl;
            return 1;
    }
    
    std::cout << std::fixed << std::setprecision(2);
    std::cout << num1 << " " << operation << " " << num2 << " = " << result << std::endl;
    
    return 0;
}

Common Mistakes

  • Integer division when float division is needed
  • Not handling division by zero
  • Ignoring overflow conditions
  • Mixing signed and unsigned types
  • Precision loss with floating-point operations

Best Practices

  • Use appropriate data types for your range and precision needs
  • Always check for division by zero
  • Use double for most floating-point calculations
  • Be aware of integer overflow
  • Use meaningful variable names for numbers
  • Format output appropriately for user readability
