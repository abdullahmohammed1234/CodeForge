C++ Strings

Strings in C++ are sequences of characters used to represent text. C++ provides two main ways to work with strings: C-style strings (character arrays) and the std::string class. Understanding both approaches is important for effective string manipulation in C++.

C-Style Strings

C-style strings are arrays of characters terminated by a null character ('\0'):

// Declaration and initialization
char greeting[] = "Hello, World!";
char name[50] = "John";

// Manual null termination
char manual[6] = {'H', 'e', 'l', 'l', 'o', '\0'};

// Character access
std::cout << greeting[0] << std::endl;  // 'H'
std::cout << greeting[7] << std::endl;  // 'W'

C String Functions

The <cstring> header provides functions for C-style strings:

#include <iostream>
#include <cstring>

int main() {
    char str1[20] = "Hello";
    char str2[20] = "World";
    char result[40];
    
    // Copy string
    strcpy(result, str1);
    std::cout << "After strcpy: " << result << std::endl;
    
    // Concatenate strings
    strcat(result, " ");
    strcat(result, str2);
    std::cout << "After strcat: " << result << std::endl;
    
    // String length
    std::cout << "Length: " << strlen(result) << std::endl;
    
    // String comparison
    char s1[] = "apple";
    char s2[] = "banana";
    int cmp = strcmp(s1, s2);
    if (cmp < 0) {
        std::cout << s1 << " comes before " << s2 << std::endl;
    }
    
    return 0;
}

std::string Class

The std::string class provides a more convenient and safer way to work with strings:

#include <iostream>
#include <string>

int main() {
    // Declaration and initialization
    std::string greeting = "Hello, World!";
    std::string name("John");
    std::string empty_str;
    
    // String operations
    std::string first = "Hello";
    std::string second = "World";
    
    // Concatenation
    std::string combined = first + " " + second;
    std::cout << combined << std::endl;
    
    // Length
    std::cout << "Length: " << combined.length() << std::endl;
    
    // Character access
    std::cout << "First char: " << combined[0] << std::endl;
    std::cout << "Last char: " << combined.back() << std::endl;
    
    return 0;
}

String Manipulation

std::string provides many useful member functions:

#include <iostream>
#include <string>

int main() {
    std::string text = "Hello, World!";
    
    // Substring
    std::string sub = text.substr(7, 5);  // "World"
    std::cout << "Substring: " << sub << std::endl;
    
    // Find
    size_t pos = text.find("World");
    if (pos != std::string::npos) {
        std::cout << "'World' found at position: " << pos << std::endl;
    }
    
    // Replace
    std::string replaced = text;
    replaced.replace(7, 5, "Universe");
    std::cout << "Replaced: " << replaced << std::endl;
    
    // Insert
    std::string inserted = text;
    inserted.insert(5, " beautiful");
    std::cout << "Inserted: " << inserted << std::endl;
    
    // Erase
    std::string erased = text;
    erased.erase(5, 7);  // Remove ", World"
    std::cout << "Erased: " << erased << std::endl;
    
    return 0;
}

String Conversion

#include <iostream>
#include <string>

int main() {
    // Number to string
    int num = 42;
    std::string str_num = std::to_string(num);
    std::cout << "Number as string: " << str_num << std::endl;
    
    double pi = 3.14159;
    std::string str_pi = std::to_string(pi);
    std::cout << "Pi as string: " << str_pi << std::endl;
    
    // String to number
    std::string num_str = "123";
    int converted = std::stoi(num_str);
    std::cout << "String to int: " << converted << std::endl;
    
    std::string float_str = "3.14";
    double converted_float = std::stod(float_str);
    std::cout << "String to double: " << converted_float << std::endl;
    
    return 0;
}

String Streams

String streams allow treating strings like I/O streams:

#include <iostream>
#include <string>
#include <sstream>

int main() {
    // Output string stream
    std::ostringstream oss;
    oss << "The answer is " << 42 << " and PI is " << 3.14;
    std::string result = oss.str();
    std::cout << result << std::endl;
    
    // Input string stream
    std::string data = "42 3.14 Hello";
    std::istringstream iss(data);
    int num;
    double pi;
    std::string word;
    iss >> num >> pi >> word;
    std::cout << "Parsed: " << num << ", " << pi << ", " << word << std::endl;
    
    return 0;
}

Example Program

#include <iostream>
#include <string>
#include <algorithm>

int main() {
    std::string text;
    std::cout << "Enter a sentence: ";
    std::getline(std::cin, text);
    
    // Count words
    std::istringstream iss(text);
    std::string word;
    int word_count = 0;
    while (iss >> word) {
        word_count++;
    }
    std::cout << "Word count: " << word_count << std::endl;
    
    // Reverse the string
    std::string reversed = text;
    std::reverse(reversed.begin(), reversed.end());
    std::cout << "Reversed: " << reversed << std::endl;
    
    // Check if palindrome
    std::string clean_text = text;
    // Remove spaces and punctuation (simplified)
    clean_text.erase(std::remove_if(clean_text.begin(), clean_text.end(), 
                   [](char c) { return !isalnum(c); }), clean_text.end());
    std::transform(clean_text.begin(), clean_text.end(), clean_text.begin(), ::tolower);
    
    std::string reversed_clean = clean_text;
    std::reverse(reversed_clean.begin(), reversed_clean.end());
    
    if (clean_text == reversed_clean) {
        std::cout << "The text is a palindrome!" << std::endl;
    } else {
        std::cout << "The text is not a palindrome." << std::endl;
    }
    
    return 0;
}

Common Mistakes

  • Forgetting null terminator in C-strings
  • Buffer overflow with C-string functions
  • Mixing cin >> and getline() without clearing input buffer
  • Using == for string comparison instead of proper methods
  • Not including necessary headers

Best Practices

  • Use std::string for most string operations
  • Use getline() for reading lines with spaces
  • Check string::npos when using find() functions
  • Use string streams for complex string building
  • Be aware of string copying performance implications
  • Use const references for string parameters in functions
