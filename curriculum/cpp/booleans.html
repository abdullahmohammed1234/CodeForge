Booleans

Boolean values in C++ represent truth values and are fundamental to decision-making in programs. C++ provides a dedicated bool data type that can hold only two values: true and false. Boolean expressions are used extensively in conditional statements, loops, and logical operations.

The bool Data Type

In C++, bool is a built-in data type that represents boolean values:

  • true: Represents a logically true condition
  • false: Represents a logically false condition
  • Internally stored as integers (true = 1, false = 0)
  • Size is typically 1 byte
#include <iostream>

int main() {
    bool is_ready = true;
    bool is_finished = false;
    
    std::cout << "is_ready: " << is_ready << std::endl;     // 1
    std::cout << "is_finished: " << is_finished << std::endl; // 0
    
    // Using std::boolalpha to print true/false instead of 1/0
    std::cout << std::boolalpha;
    std::cout << "is_ready: " << is_ready << std::endl;     // true
    std::cout << "is_finished: " << is_finished << std::endl; // false
    
    return 0;
}

Boolean Literals and Keywords

C++ uses the keywords true and false as boolean literals:

// Boolean literals
bool flag1 = true;
bool flag2 = false;

// Boolean expressions evaluate to true or false
bool result1 = (5 > 3);    // true
bool result2 = (10 < 5);   // false
bool result3 = (7 == 7);   // true
bool result4 = (8 != 8);   // false

Comparison Operators

Comparison operators return boolean values:

#include <iostream>

int main() {
    int a = 10, b = 20;
    
    std::cout << std::boolalpha;
    std::cout << "a == b: " << (a == b) << std::endl;  // false
    std::cout << "a != b: " << (a != b) << std::endl;  // true
    std::cout << "a < b: " << (a < b) << std::endl;    // true
    std::cout << "a > b: " << (a > b) << std::endl;    // false
    std::cout << "a <= b: " << (a <= b) << std::endl;  // true
    std::cout << "a >= b: " << (a >= b) << std::endl;  // false
    
    return 0;
}

Logical Operators

Logical operators combine boolean expressions:

  • && (AND): True if both operands are true
  • || (OR): True if at least one operand is true
  • ! (NOT): Negates the operand
#include <iostream>

int main() {
    bool x = true, y = false;
    
    std::cout << std::boolalpha;
    std::cout << "x && y: " << (x && y) << std::endl;  // false
    std::cout << "x || y: " << (x || y) << std::endl;  // true
    std::cout << "!x: " << (!x) << std::endl;          // false
    std::cout << "!y: " << (!y) << std::endl;          // true
    
    // Short-circuit evaluation
    bool result = (x && y) || (x || y);  // true
    std::cout << "Complex expression: " << result << std::endl;
    
    return 0;
}

Boolean Conversion

Non-boolean values can be converted to boolean:

#include <iostream>

int main() {
    std::cout << std::boolalpha;
    
    // Numeric to boolean conversion
    int num1 = 0, num2 = 5;
    double decimal = 0.0;
    
    std::cout << "bool(0): " << bool(num1) << std::endl;        // false
    std::cout << "bool(5): " << bool(num2) << std::endl;        // true
    std::cout << "bool(0.0): " << bool(decimal) << std::endl;   // false
    
    // Pointer to boolean conversion
    int* ptr1 = nullptr;
    int* ptr2 = new int(42);
    
    std::cout << "bool(nullptr): " << bool(ptr1) << std::endl;  // false
    std::cout << "bool(valid_ptr): " << bool(ptr2) << std::endl; // true
    
    delete ptr2;
    
    return 0;
}

Conditional Statements

Boolean expressions control program flow:

#include <iostream>

int main() {
    int age = 25;
    bool has_license = true;
    
    // if statement
    if (age >= 18 && has_license) {
        std::cout << "You can drive!" << std::endl;
    } else {
        std::cout << "You cannot drive." << std::endl;
    }
    
    // Ternary operator
    std::string status = (age >= 18) ? "Adult" : "Minor";
    std::cout << "Status: " << status << std::endl;
    
    return 0;
}

Boolean in Loops

#include <iostream>

int main() {
    bool continue_loop = true;
    int counter = 0;
    
    while (continue_loop) {
        std::cout << "Iteration: " << counter << std::endl;
        counter++;
        
        if (counter >= 5) {
            continue_loop = false;  // Exit condition
        }
    }
    
    std::cout << "Loop finished!" << std::endl;
    
    return 0;
}

Boolean Functions

Functions can return boolean values:

#include <iostream>
#include <string>

bool is_even(int number) {
    return number % 2 == 0;
}

bool is_valid_age(int age) {
    return age >= 0 && age <= 150;
}

bool contains_substring(const std::string& text, const std::string& substring) {
    return text.find(substring) != std::string::npos;
}

int main() {
    std::cout << std::boolalpha;
    
    std::cout << "is_even(4): " << is_even(4) << std::endl;
    std::cout << "is_even(7): " << is_even(7) << std::endl;
    std::cout << "is_valid_age(25): " << is_valid_age(25) << std::endl;
    std::cout << "is_valid_age(-5): " << is_valid_age(-5) << std::endl;
    
    std::string text = "Hello, World!";
    std::cout << "contains 'World': " << contains_substring(text, "World") << std::endl;
    std::cout << "contains 'Python': " << contains_substring(text, "Python") << std::endl;
    
    return 0;
}

Common Boolean Pitfalls

  • Assignment (=) vs comparison (==)
  • Logical operators vs bitwise operators (& vs &&, | vs ||)
  • Floating-point comparisons
  • Null pointer checks
// Common mistakes
int x = 5;

// Wrong: assignment in condition
// if (x = 10) { ... }  // Always true, assigns 10 to x

// Correct: comparison
if (x == 10) { /* ... */ }

// Wrong: bitwise AND instead of logical AND
// if (x > 0 & y > 0) { ... }  // Bitwise operation

// Correct: logical AND
if (x > 0 && y > 0) { /* ... */ }

Mini Project

Create a program that validates user input using boolean functions:

  • Create boolean functions to validate different types of input
  • Implement email validation, password strength checking, and number range validation
  • Use boolean logic to combine multiple validation conditions
  • Provide clear feedback to the user about validation results
  • Bonus: Create a user registration system with multiple validation steps

Here's a starting template:

#include <iostream>
#include <string>
#include <cctype>

bool is_valid_email(const std::string& email) {
    // Basic email validation
    size_t at_pos = email.find('@');
    size_t dot_pos = email.find('.', at_pos);
    
    return at_pos != std::string::npos &&
           dot_pos != std::string::npos &&
           at_pos > 0 &&
           dot_pos > at_pos + 1;
}

bool is_strong_password(const std::string& password) {
    bool has_upper = false, has_lower = false, has_digit = false;
    int length = password.length();
    
    if (length < 8) return false;
    
    for (char c : password) {
        if (std::isupper(c)) has_upper = true;
        else if (std::islower(c)) has_lower = true;
        else if (std::isdigit(c)) has_digit = true;
    }
    
    return has_upper && has_lower && has_digit;
}

bool is_in_range(int value, int min_val, int max_val) {
    return value >= min_val && value <= max_val;
}

int main() {
    std::string email, password;
    int age;
    
    std::cout << "User Validation System" << std::endl;
    
    // Email validation
    do {
        std::cout << "Enter email: ";
        std::getline(std::cin, email);
    } while (!is_valid_email(email));
    
    // Password validation
    do {
        std::cout << "Enter password (min 8 chars, upper+lower+digit): ";
        std::getline(std::cin, password);
    } while (!is_strong_password(password));
    
    // Age validation
    do {
        std::cout << "Enter age (18-100): ";
        std::cin >> age;
    } while (!is_in_range(age, 18, 100));
    
    std::cout << "All validations passed! Registration successful." << std::endl;
    
    return 0;
}
