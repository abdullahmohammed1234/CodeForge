Dates and Times

Working with dates and times is essential for many applications, from logging events to scheduling tasks. Python's datetime module provides comprehensive tools for handling dates, times, and time intervals. Understanding these concepts allows you to create time-aware applications, perform date calculations, and format temporal data appropriately.

Why Dates and Times Matter

Date and time handling is crucial for:

  • Logging: Recording when events occur
  • Scheduling: Planning future tasks and reminders
  • Data Analysis: Analyzing trends over time
  • User Interfaces: Displaying dates in user-friendly formats
  • Business Logic: Calculating deadlines and time-based rules
  • Internationalization: Handling different time zones and locales

The datetime Module

Import the datetime module to work with dates and times:

import datetime

# Get current date and time
now = datetime.datetime.now()
print(now)  # 2024-01-05 14:30:45.123456

# Get current date only
today = datetime.date.today()
print(today)  # 2024-01-05

# Get current time only
current_time = datetime.datetime.now().time()
print(current_time)  # 14:30:45.123456

Formatting Dates and Times

Use strftime() to format datetime objects as strings:

import datetime

now = datetime.datetime.now()

# Common format codes
print(now.strftime("%Y-%m-%d"))          # 2024-01-05
print(now.strftime("%H:%M:%S"))          # 14:30:45
print(now.strftime("%A, %B %d, %Y"))     # Friday, January 05, 2024
print(now.strftime("%I:%M %p"))          # 02:30 PM

# Custom formats
print(now.strftime("%Y/%m/%d %H:%M"))    # 2024/01/05 14:30
print(now.strftime("%d-%b-%Y"))          # 05-Jan-2024

# Format codes reference:
# %Y - Year with century (2024)
# %y - Year without century (24)
# %m - Month as zero-padded decimal (01-12)
# %B - Full month name (January)
# %b - Abbreviated month name (Jan)
# %d - Day of month as zero-padded decimal (01-31)
# %A - Full weekday name (Friday)
# %a - Abbreviated weekday name (Fri)
# %H - Hour (24-hour clock) as zero-padded decimal (00-23)
# %I - Hour (12-hour clock) as zero-padded decimal (01-12)
# %M - Minute as zero-padded decimal (00-59)
# %S - Second as zero-padded decimal (00-59)
# %p - AM or PM

# Handle different formats
def parse_date_flexible(date_string):
    """Try different common date formats"""
    formats = [
        "%Y-%m-%d",
        "%m/%d/%Y",
        "%d/%m/%Y",
        "%B %d, %Y",
        "%b %d, %Y",
        "%Y/%m/%d"

import datetime

# Create timezone-aware datetime
utc_tz = datetime.timezone.utc

utc_now = datetime.datetime.now(utc_tz)

pacific_time = utc_now.astimezone(pacific_tz)

print("Pacific Time:", pacific_time)

# Create timezone-aware datetime from naive datetime
naive_time = datetime.datetime(2024, 1, 5, 12, 0, 0)
utc_aware = naive_time.replace(tzinfo=utc_tz)
print("UTC aware:", utc_aware)

# List of common timezones (you might want to use pytz library for more)

import calendar

# Get day of week (0=Monday, 6=Sunday)
date = datetime.date(2024, 1, 5)  # Friday
print(f"January 5, 2024 is a {calendar.day_name[date.weekday()]}")

# Find first Monday of month
def first_weekday_of_month(year, month, weekday):
    """Find the first occurrence of a weekday in a month"""
    # weekday: 0=Monday, 6=Sunday
    for day in range(1, 8):
        date = datetime.date(year, month, day)
        if date.weekday() == weekday:
            return date
    return None

first_monday = first_weekday_of_month(2024, 1, 0)  # Monday
print(f"First Monday of January 2024: {first_monday}")

# Calculate weeks in month
def weeks_in_month(year, month):
    """Calculate number of weeks in a month"""
    first_day = datetime.date(year, month, 1)
    last_day = datetime.date(year, month, calendar.monthrange(year, month)[1])
    
    # Calculate total days
    total_days = (last_day - first_day).days + 1
    
    # Calculate weeks (ceiling division)
    weeks = (total_days + first_day.weekday()) // 7
    if (total_days + first_day.weekday()) % 7 > 0:
        weeks += 1
    
    return weeks

print(f"Weeks in January 2024: {weeks_in_month(2024, 1)}")
print(f"Weeks in February 2024: {weeks_in_month(2024, 2)}")

Practical Examples

import datetime

def calculate_age(birth_date):
    """Calculate age from birth date"""
    today = datetime.date.today()
    age = today.year - birth_date.year
    
    # Adjust if birthday hasn't occurred this year
    if (today.month, today.day) < (birth_date.month, birth_date.day):
        age -= 1
    
    return age

def days_until_event(event_date):
    """Calculate days until an event"""
    today = datetime.date.today()
    if event_date < today:
        return "Event has passed"
    
    days = (event_date - today).days
    if days == 0:
        return "Today!"
    elif days == 1:
        return "Tomorrow"
    else:
        return f"{days} days"

def format_duration(seconds):
    """Format seconds into human-readable duration"""
    days, remainder = divmod(seconds, 86400)
    hours, remainder = divmod(remainder, 3600)
    minutes, seconds = divmod(remainder, 60)
    
    parts = []
    if days > 0:
        parts.append(f"{days} day{'s' if days != 1 else ''}")
    if hours > 0:
        parts.append(f"{hours} hour{'s' if hours != 1 else ''}")
    if minutes > 0:
        parts.append(f"{minutes} minute{'s' if minutes != 1 else ''}")
    if seconds > 0 or not parts:
        parts.append(f"{seconds} second{'s' if seconds != 1 else ''}")
    
    return ", ".join(parts)

# Test the functions
birth_date = datetime.date(1990, 5, 15)
print(f"Age: {calculate_age(birth_date)}")

christmas = datetime.date(2024, 12, 25)
print(f"Days until Christmas: {days_until_event(christmas)}")

duration_seconds = 3661  # 1 hour, 1 minute, 1 second
print(f"Duration: {format_duration(duration_seconds)}")

# Work schedule calculator
def calculate_work_hours(start_time, end_time, break_minutes=0):
    """Calculate work hours between start and end time"""
    duration = end_time - start_time
    total_minutes = duration.total_seconds() / 60
    work_minutes = total_minutes - break_minutes
    work_hours = work_minutes / 60
    
    return work_hours

start = datetime.datetime(2024, 1, 5, 9, 0, 0)  # 9:00 AM
end = datetime.datetime(2024, 1, 5, 17, 30, 0)  # 5:30 PM
work_hours = calculate_work_hours(start, end, break_minutes=30)
print(f"Work hours: {work_hours:.2f}")

Create a comprehensive date and time utility program:

  • Display current date and time in multiple formats
  • Parse dates from various string formats
  • Calculate time differences and durations

Here's a starting template:

import datetime
import calendar
from collections import defaultdict

class DateTimeUtility:
    def __init__(self):
        self.events = []
        
    def get_current_info(self):
        """Get comprehensive current date/time information"""
        now = datetime.datetime.now()
        today = datetime.date.today()
        
        info = {
            "datetime": now,
            "date": today,
            "time": now.time(),
            "timestamp": now.timestamp(),
            "weekday": calendar.day_name[today.weekday()],
            "month": calendar.month_name[today.month],
            "day_of_year": today.timetuple().tm_yday,
            "week_of_year": today.isocalendar()[1],
            "is_leap_year": calendar.isleap(today.year),
            "days_in_month": calendar.monthrange(today.year, today.month)[1]
        }
        
        return info
    
    def format_datetime(self, dt, format_type="full"):
        """Format datetime in various styles"""
        formats = {
            "full": "%A, %B %d, %Y at %I:%M:%S %p",
            "date": "%Y-%m-%d",
            "time": "%H:%M:%S",
            "iso": "%Y-%m-%dT%H:%M:%S",
            "us": "%m/%d/%Y",
            "european": "%d/%m/%Y",
            "long": "%A, %B %d, %Y",
            "short": "%b %d, %Y"
        }
        
        return dt.strftime(formats.get(format_type, formats["full"]))
    
    def parse_date_string(self, date_string):
        """Parse various date string formats"""
        formats = [
            "%Y-%m-%d",
            "%m/%d/%Y",
            "%d/%m/%Y",
            "%B %d, %Y",
            "%b %d, %Y",
            "%Y/%m/%d",
            "%d-%b-%Y",
            "%Y%m%d"
        ]
        
        for fmt in formats:
            try:
                if " " in date_string and ":" in date_string:
                    # Try datetime parsing
                    return datetime.datetime.strptime(date_string, fmt + " %H:%M:%S")
                else:
                    # Try date parsing
                    return datetime.datetime.strptime(date_string, fmt).date()
            except ValueError:
                continue
        
        raise ValueError(f"Unable to parse date: {date_string}")
    
    def calculate_age(self, birth_date):
        """Calculate age from birth date"""
        today = datetime.date.today()
        
        if isinstance(birth_date, str):
            birth_date = self.parse_date_string(birth_date)
        
        if isinstance(birth_date, datetime.datetime):
            birth_date = birth_date.date()
        
        age = today.year - birth_date.year
        
        # Adjust if birthday hasn't occurred this year
        if (today.month, today.day) < (birth_date.month, birth_date.day):
            age -= 1
        
        return age
    
    def time_difference(self, start_date, end_date=None):
        """Calculate time difference between dates"""
        if end_date is None:
            end_date = datetime.datetime.now()
        
        if isinstance(start_date, str):
            start_date = self.parse_date_string(start_date)
        if isinstance(end_date, str):
            end_date = self.parse_date_string(end_date)
        
        # Ensure both are datetime objects
        if isinstance(start_date, datetime.date) and not isinstance(start_date, datetime.datetime):
            start_date = datetime.datetime.combine(start_date, datetime.time.min)
        if isinstance(end_date, datetime.date) and not isinstance(end_date, datetime.datetime):
            end_date = datetime.datetime.combine(end_date, datetime.time.min)
        
        diff = end_date - start_date
        
        return {
            "total_seconds": diff.total_seconds(),
            "days": diff.days,
            "hours": diff.seconds // 3600,
            "minutes": (diff.seconds % 3600) // 60,
            "seconds": diff.seconds % 60
        }
    
    def add_event(self, name, date, description=""):
        """Add an event to the scheduler"""
        if isinstance(date, str):
            date = self.parse_date_string(date)
        
        event = {
            "name": name,
            "date": date,
            "description": description
        }
        
        self.events.append(event)
        self.events.sort(key=lambda x: x["date"])
    
    def get_upcoming_events(self, days_ahead=7):
        """Get events happening in the next N days"""
        now = datetime.datetime.now()
        future_date = now + datetime.timedelta(days=days_ahead)
        
        upcoming = []
        for event in self.events:
            event_datetime = event["date"]
            if isinstance(event_datetime, datetime.date) and not isinstance(event_datetime, datetime.datetime):
                event_datetime = datetime.datetime.combine(event_datetime, datetime.time.min)
            
            if now <= event_datetime <= future_date:
                upcoming.append(event)
        
        return upcoming
    
    def create_calendar_view(self, year, month):
        """Create a simple calendar view for a month"""
        cal = calendar.monthcalendar(year, month)
        
        # Get month name
        month_name = calendar.month_name[month]
        
        # Create header
        header = f"{month_name} {year}"
        header = header.center(20)
        
        # Create calendar lines
        lines = [header, "Mo Tu We Th Fr Sa Su"]
        
        for week in cal:
            week_str = ""
            for day in week:
                if day == 0:
                    week_str += "   "
                else:
                    week_str += f"{day:2d} "
            lines.append(week_str.rstrip())
        
        return "\n".join(lines)
    
    def validate_date(self, year, month, day):
        """Validate if a date is valid"""
        try:
            datetime.date(year, month, day)
            return True
        except ValueError:
            return False

Common Mistakes

  • Confusing naive and aware datetime objects
  • Not handling time zone conversions properly
  • Using wrong format codes in strftime/strptime
  • Forgetting that months are 1-based but days are 1-based
  • Mixing date and datetime objects inappropriately
  • Not accounting for leap years in calculations
  • Using timestamps without considering time zones

Best Practices

  • Use timezone-aware datetime objects when possible
  • Store dates as UTC internally and convert for display
  • Validate date inputs before processing
  • Use consistent date formats throughout your application
  • Consider using dateutil or arrow libraries for complex operations
  • Handle daylight saving time changes carefully
  • Document expected date formats in your code
