Sets

Sets in Python are unordered collections of unique elements. They are useful for storing items where order doesn't matter and duplicates are not allowed. Sets support mathematical set operations like union, intersection, and difference.

What are Sets?

  • Unordered collections of unique elements
  • Mutable (can be modified after creation)
  • Do not allow duplicate values
  • Elements must be hashable (immutable)
  • No indexing or slicing
  • Support mathematical set operations

Creating Sets

# Empty set
empty_set = set()

# Set with elements
fruits = {"apple", "banana", "cherry"}
numbers = {1, 2, 3, 4, 5}

# Set from list (removes duplicates)
list_with_duplicates = [1, 2, 2, 3, 3, 3]
unique_numbers = set(list_with_duplicates)  # {1, 2, 3}

# Set from string
letters = set("hello")  # {'h', 'e', 'l', 'o'}

print(fruits)      # {'cherry', 'banana', 'apple'} (order may vary)
print(numbers)     # {1, 2, 3, 4, 5}
print(unique_numbers)  # {1, 2, 3}
print(letters)     # {'h', 'e', 'l', 'o'}

Set Operations

Adding Elements

fruits = {"apple", "banana"}

# Add single element
fruits.add("cherry")
print(fruits)  # {'cherry', 'banana', 'apple'}

# Add multiple elements
fruits.update(["orange", "grape"])
print(fruits)  # {'cherry', 'banana', 'apple', 'orange', 'grape'}

Removing Elements

fruits = {"apple", "banana", "cherry", "orange"}

# Remove specific element (raises KeyError if not found)
fruits.remove("banana")
print(fruits)  # {'cherry', 'apple', 'orange'}

# Remove element if it exists (no error if not found)
fruits.discard("grape")  # No error
fruits.discard("apple")
print(fruits)  # {'cherry', 'orange'}

# Remove and return arbitrary element
removed = fruits.pop()
print(f"Removed: {removed}")  # Random element
print(fruits)

# Clear all elements
fruits.clear()
print(fruits)  # set()

Set Membership

fruits = {"apple", "banana", "cherry"}

print("apple" in fruits)     # True
print("grape" in fruits)     # False
print("apple" not in fruits) # False

Mathematical Set Operations

set_a = {1, 2, 3, 4}
set_b = {3, 4, 5, 6}

# Union: elements in either set
print("Union:", set_a | set_b)           # {1, 2, 3, 4, 5, 6}
print("Union:", set_a.union(set_b))      # {1, 2, 3, 4, 5, 6}

# Intersection: elements in both sets
print("Intersection:", set_a & set_b)    # {3, 4}
print("Intersection:", set_a.intersection(set_b))  # {3, 4}

# Difference: elements in set_a but not in set_b
print("Difference:", set_a - set_b)      # {1, 2}
print("Difference:", set_a.difference(set_b))  # {1, 2}

# Symmetric difference: elements in either set but not both
print("Symmetric diff:", set_a ^ set_b)  # {1, 2, 5, 6}
print("Symmetric diff:", set_a.symmetric_difference(set_b))  # {1, 2, 5, 6}

Set Comparisons

set_a = {1, 2, 3}
set_b = {1, 2, 3, 4}
set_c = {2, 3}

# Subset: all elements of set_a are in set_b
print(set_a <= set_b)  # True (set_a is subset of set_b)
print(set_a.issubset(set_b))  # True

# Proper subset: subset and not equal
print(set_a < set_b)   # True

# Superset: set_b contains all elements of set_a
print(set_b >= set_a)  # True
print(set_b.issuperset(set_a))  # True

# Proper superset
print(set_b > set_a)   # True

# Disjoint: no common elements
print(set_a.isdisjoint({4, 5, 6}))  # True

Frozen Sets

Frozen sets are immutable versions of sets:

# Create frozen set
frozen = frozenset([1, 2, 3, 4])

print(frozen)  # frozenset({1, 2, 3, 4})

# Can be used in set operations
regular_set = {3, 4, 5}
print(frozen & regular_set)  # frozenset({3, 4})

# Cannot be modified
# frozen.add(5)  # AttributeError

Set Comprehensions

Set comprehensions create sets using concise syntax:

# Create set of squares
squares = {x**2 for x in range(1, 6)}
print(squares)  # {1, 4, 9, 16, 25}

# Set comprehension with condition
even_squares = {x**2 for x in range(1, 11) if x % 2 == 0}
print(even_squares)  # {4, 16, 36, 64, 100}

# Set from string (unique characters)
unique_chars = {char for char in "hello world" if char != " "}
print(unique_chars)  # {'h', 'e', 'l', 'o', 'w', 'r', 'd'}

Common Use Cases

  • Removing duplicates from a list
  • Membership testing (faster than lists)
  • Finding unique elements
  • Mathematical set operations
  • Removing common elements between collections

Mini Project

Create a program that analyzes text using sets:

  • Ask the user to input a sentence
  • Convert the sentence to a set of unique words
  • Find common words between two sentences
  • Find words unique to each sentence
  • Display statistics about word usage
  • Bonus: Implement case-insensitive comparison

Here's a starting template:

def analyze_text():
    # Get user input
    sentence1 = input("Enter first sentence: ").lower()
    sentence2 = input("Enter second sentence: ").lower()
    
    # Convert to sets of words
    words1 = set(sentence1.split())
    words2 = set(sentence2.split())
    
    print(f"Unique words in sentence 1: {words1}")
    print(f"Unique words in sentence 2: {words2}")
    
    # Set operations
    common_words = words1 & words2
    unique_to_first = words1 - words2
    unique_to_second = words2 - words1
    all_words = words1 | words2
    
    print(f"Common words: {common_words}")
    print(f"Unique to first: {unique_to_first}")
    print(f"Unique to second: {unique_to_second}")
    print(f"All unique words: {all_words}")
    print(f"Total unique words: {len(all_words)}")

analyze_text()
