String Formatting

String formatting in Python allows you to create formatted strings by embedding values into templates. Python provides several methods for string formatting, each with different syntax and capabilities.

Why String Formatting?

  • Create readable output with embedded values
  • Control number formatting (decimals, padding, etc.)
  • Handle different data types automatically
  • Improve code maintainability
  • Internationalization support

Old Style Formatting (% Operator)

# Basic formatting
name = "Alice"
age = 25
print("My name is %s and I am %d years old." % (name, age))

# Number formatting
pi = 3.14159265359
print("Pi to 2 decimal places: %.2f" % pi)
print("Pi to 4 decimal places: %.4f" % pi)

# Width and alignment
print("Right aligned: %10s" % "hello")
print("Left aligned: %-10s" % "hello")

# Multiple values
x, y, z = 1, 2, 3
print("Values: %d, %d, %d" % (x, y, z))

# Dictionary formatting
person = {"name": "Bob", "age": 30}
print("Name: %(name)s, Age: %(age)d" % person)

str.format() Method

# Positional arguments
print("My name is {} and I am {} years old.".format("Alice", 25))

# Indexed arguments
print("My name is {0} and I am {1} years old.".format("Alice", 25))
print("I am {1} years old and my name is {0}.".format("Alice", 25))

# Named arguments
print("My name is {name} and I am {age} years old.".format(name="Alice", age=25))

# Dictionary arguments
person = {"name": "Bob", "age": 30}
print("My name is {name} and I am {age} years old.".format(**person))

# Number formatting
pi = 3.14159265359
print("Pi: {:.2f}".format(pi))
print("Pi: {:.4f}".format(pi))

# Width and alignment
print("Right aligned: {:>10}".format("hello"))
print("Left aligned: {:<10}".format("hello"))
print("Centered: {:^10}".format("hello"))

# Fill character
print("Filled: {:*>10}".format("hello"))

f-Strings (Python 3.6+)

f-strings provide the most readable and efficient string formatting:

name = "Alice"
age = 25

# Basic f-string
print(f"My name is {name} and I am {age} years old.")

# Expressions in f-strings
print(f"Next year I will be {age + 1} years old.")
print(f"My name in uppercase: {name.upper()}")

# Number formatting
pi = 3.14159265359
print(f"Pi to 2 decimal places: {pi:.2f}")
print(f"Pi to 4 decimal places: {pi:.4f}")

# Width and alignment
print(f"Right aligned: {name:>10}")
print(f"Left aligned: {name:<10}")
print(f"Centered: {name:^10}")

# Number formatting with commas
large_number = 1234567
print(f"Large number: {large_number:,}")

# Binary, octal, hexadecimal
num = 42
print(f"Decimal: {num}")
print(f"Binary: {num:b}")
print(f"Octal: {num:o}")
print(f"Hexadecimal: {num:x}")

# Date formatting
from datetime import datetime
now = datetime.now()
print(f"Current date: {now:%Y-%m-%d}")
print(f"Current time: {now:%H:%M:%S}")

Advanced Formatting Options

# Thousands separator
number = 1234567.89
print(f"With commas: {number:,.2f}")

# Percentage
ratio = 0.856
print(f"Percentage: {ratio:.1%}")

# Scientific notation
small_number = 0.00001234
print(f"Scientific: {small_number:.2e}")

# Fixed width with sign
values = [42, -17, 0]
for val in values:
    print(f"Value: {val:+5d}")

# Zero padding
for i in range(1, 6):
    print(f"File {i:03d}.txt")

Formatting Objects

class Person:
    def __init__(self, name, age):
        self.name = name
        self.age = age
    
    def __str__(self):
        return f"Person(name='{self.name}', age={self.age})"
    
    def __repr__(self):
        return f"Person('{self.name}', {self.age})"

person = Person("Alice", 25)
print(f"Person: {person}")
print(f"Representation: {person!r}")

# Custom formatting
class Temperature:
    def __init__(self, celsius):
        self.celsius = celsius
    
    def __format__(self, format_spec):
        if format_spec == "F":
            return f"{self.celsius * 9/5 + 32:.1f}°F"
        elif format_spec == "K":
            return f"{self.celsius + 273.15:.1f}K"
        else:
            return f"{self.celsius:.1f}°C"

temp = Temperature(20)
print(f"Temperature: {temp}")
print(f"Fahrenheit: {temp:F}")
print(f"Kelvin: {temp:K}")

String Formatting Best Practices

  • Use f-strings for Python 3.6+ (most readable and efficient)
  • Use .format() for complex formatting or older Python versions
  • Avoid % formatting for new code
  • Be consistent within a project
  • Use descriptive variable names in format strings
  • Consider internationalization when designing format strings

Mini Project

Create a report generator using string formatting:

  • Create a list of student records with names, grades, and attendance
  • Format a report showing individual student statistics
  • Calculate and display class averages
  • Use different formatting techniques (f-strings, .format(), %)
  • Include proper alignment and number formatting
  • Bonus: Add grade letter conversion (A, B, C, etc.)

Here's a starting template:

def calculate_grade_letter(score):
    """Convert numeric score to letter grade."""
    if score >= 90:
        return 'A'
    elif score >= 80:
        return 'B'
    elif score >= 70:
        return 'C'
    elif score >= 60:
        return 'D'
    else:
        return 'F'

def generate_student_report(students):
    """Generate formatted report for students."""
    print("=" * 60)
    print(f"{'Student Report':^60}")
    print("=" * 60)
    print(f"{'Name':<20} {'Grade':<8} {'Letter':<8} {'Attendance':<12}")
    print("-" * 60)
    
    total_score = 0
    total_attendance = 0
    
    for student in students:
        name, score, attendance = student
        grade_letter = calculate_grade_letter(score)
        
        # Using f-strings for formatting
        print(f"{name:<20} {score:<8.1f} {grade_letter:<8} {attendance:<12.1f}%")
        
        total_score += score
        total_attendance += attendance
    
    print("-" * 60)
    
    avg_score = total_score / len(students)
    avg_attendance = total_attendance / len(students)
    
    # Using .format() method
    print("Class Average: {:.1f} ({})".format(avg_score, calculate_grade_letter(avg_score)))
    print("Average Attendance: {:.1f}%".format(avg_attendance))

# Sample data
students = [
    ("Alice Johnson", 95.5, 98.5),
    ("Bob Smith", 87.2, 92.1),
    ("Charlie Brown", 78.9, 85.3),
    ("Diana Prince", 91.7, 96.8),
    ("Eve Wilson", 83.4, 89.2)
]

generate_student_report(students)
