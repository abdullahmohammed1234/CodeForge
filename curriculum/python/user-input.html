User Input

User input allows your Python programs to interact with users by accepting data during runtime. Python provides built-in functions to read input from the keyboard, making programs more dynamic and interactive.

The input() Function

The primary function for reading user input:

# Basic input
name = input("Enter your name: ")
print(f"Hello, {name}!")

# Input without prompt
age = input()
print(f"You entered: {age}")

# Input with multi-line prompt
user_input = input("""Enter something:
> """)
print(f"You typed: {user_input}")

Input Data Types

input() always returns a string. Convert to other types as needed:

# String input (default)
name = input("Enter your name: ")
print(type(name))  # 

# Integer input
age = int(input("Enter your age: "))
print(f"Next year you will be {age + 1}")

# Float input
height = float(input("Enter your height in meters: "))
print(f"Your height is {height:.2f} meters")

# Boolean input
is_student = input("Are you a student? (yes/no): ").lower() == "yes"
print(f"Student status: {is_student}")

Input Validation

Handle invalid input gracefully:

# Basic validation with try-except
def get_valid_age():
    while True:
        try:
            age = int(input("Enter your age: "))
            if age < 0:
                print("Age cannot be negative. Try again.")
                continue
            if age > 150:
                print("Age seems too high. Try again.")
                continue
            return age
        except ValueError:
            print("Please enter a valid number.")

age = get_valid_age()
print(f"Your age is: {age}")

# Validation with conditional checks
def get_yes_no(prompt):
    while True:
        response = input(prompt).lower().strip()
        if response in ['yes', 'y', 'true', '1']:
            return True
        elif response in ['no', 'n', 'false', '0']:
            return False
        else:
            print("Please enter yes or no.")

likes_python = get_yes_no("Do you like Python? ")
print(f"Likes Python: {likes_python}")

Multiple Input Values

# Input with custom separator

# Input with timeout (requires signal module)
import signal

def timeout_handler(signum, frame):
    raise TimeoutError("Input timeout")

def input_with_timeout(prompt, timeout=5):
    signal.signal(signal.SIGALRM, timeout_handler)
    signal.alarm(timeout)
    try:
        result = input(prompt)
        signal.alarm(0)  # Cancel alarm
        return result
    except TimeoutError:
        print("\nInput timeout!")
        return None

Input Buffering and Issues

# Issue: newline character left in buffer
print("Enter your name:")
name = input()
print("Enter your age:")
age = input()  # This will work fine

# Better approach: use single input calls
name = input("Enter your name: ")
age = input("Enter your age: ")

# Handling empty input
user_input = input("Enter something (or press Enter for default): ").strip()
if not user_input:
    user_input = "default value"
print(f"Input: '{user_input}'")

# Input with default values
def get_input_with_default(prompt, default=""):
    user_input = input(f"{prompt} (default: {default}): ").strip()
    return user_input if user_input else default

name = get_input_with_default("Enter your name", "Anonymous")
print(f"Hello, {name}!")

Input from Files

# Reading from file instead of keyboard
# Redirect stdin from file when running: python script.py < input.txt

# Or read file directly
def read_from_file(filename):
    try:
        with open(filename, 'r') as file:
            content = file.read()
            return content
    except FileNotFoundError:
        print(f"File {filename} not found")
        return None

Mini Project

Create an interactive quiz program:

  • Create a list of questions with multiple choice answers
  • Prompt user for answers and validate input
  • Keep track of score and provide feedback
  • Handle invalid inputs gracefully
  • Display final results with percentage
  • Bonus: Add timer functionality and save results to file

Here's a starting template:

def get_valid_choice(options):
    """Get a valid choice from user."""
    while True:
        try:
            choice = int(input("Your choice (1-4): "))
            if 1 <= choice <= len(options):
                return choice - 1  # Convert to 0-based index
            else:
                print(f"Please enter a number between 1 and {len(options)}")
        except ValueError:
            print("Please enter a valid number")

def run_quiz():
    """Run the interactive quiz."""
    questions = [
        {
            "question": "What is the capital of France?",
            "options": ["London", "Berlin", "Paris", "Madrid"],
            "correct": 2
        },
        {
            "question": "Which planet is known as the Red Planet?",
            "options": ["Venus", "Mars", "Jupiter", "Saturn"],
            "correct": 1
        },
        {
            "question": "What is 2 + 2?",
            "options": ["3", "4", "5", "6"],
            "correct": 1
        }
    ]
    
    score = 0
    
    print("Welcome to the Quiz!")
    print("=" * 30)
    
    for i, q in enumerate(questions, 1):
        print(f"\nQuestion {i}: {q['question']}")
        for j, option in enumerate(q['options'], 1):
            print(f"{j}. {option}")
        
        user_choice = get_valid_choice(q['options'])
        
        if user_choice == q['correct']:
            print("Correct!")
            score += 1
        else:
            correct_answer = q['options'][q['correct']]
            print(f"Incorrect. The correct answer is: {correct_answer}")
    
    print(f"\nQuiz completed!")
    print(f"Your score: {score}/{len(questions)}")
    percentage = (score / len(questions)) * 100
    print(f"Percentage: {percentage:.1f}%")
    
    if percentage >= 80:
        print("Excellent work!")
    elif percentage >= 60:
        print("Good job!")
    else:
        print("Keep practicing!")

# Run the quiz
run_quiz()
